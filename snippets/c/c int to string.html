#include <stdio.h>



char *my_itoa(int a,char *buffer, int buff_size)
{
  char *temp;
  int b;
  temp = buffer;

  b = a;
  if (buffer == NULL) 
     return NULL;

  while (a > 0)
  {
     if (buff_size < 1)
        return NULL;

     b = a%10;
     *temp = b + '0' ;
     temp++;
     a = a/10;
     buff_size--;
  }
  *temp = '\0';
  /*Note : You have to Add a String reverse Function on this buffer here*/
  return buffer;
}
